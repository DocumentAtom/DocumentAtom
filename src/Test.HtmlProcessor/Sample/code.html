JavaScript Example


function greet(name) {
    console.log(`Hello, ${name}!`);
    return name.toUpperCase();
}

const result = greet("World");
console.log(result);

Python Example


def factorial(n):
    if n <= 1:
        return 1
    return n * factorial(n - 1)

print(factorial(5))
